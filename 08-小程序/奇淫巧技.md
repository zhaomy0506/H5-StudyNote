```scss
// 页面  <swiper class="square-dot" indicator-dots indicator-active-color="#fff">...</swiper>

    background-color: #000;
    opacity: 0.4;
    width: 10rpx;
    height: 10rpx;
    border-radius: 20rpx;
    margin: 0 8rpx !important;
}

swiper.square-dot .wx-swiper-dot.wx-swiper-dot-active{
    opacity: 1;
    width: 30rpx;
}
```
